11    pub fn assign_digest_cells(
12        &self,
13        region: &mut Region<'_, pallas::Base>,
14        state: State,
15    ) -> Result<[AssignedWord32; DIGEST_SIZE], Error> {
16        let a_3 = self.extras[0];
17        let a_4 = self.extras[1];
18        let a_5 = self.message_schedule;
19        let a_6 = self.extras[2];
20        let a_7 = self.extras[3];
21        let a_8 = self.extras[4];
22
23        let (a, b, c, d, e, f, g, h) = match_state(state);
24
25        let abcd_row = 0;
26        self.s_digest.enable(region, abcd_row)?;
27        let efgh_row = abcd_row + 2;
28        self.s_digest.enable(region, efgh_row)?;
29
30        // Assign digest for A, B, C, D
31        a.dense_halves
32            .0
33            .copy_advice(|| "a_lo", region, a_3, abcd_row)?;
34        a.dense_halves
35            .1
36            .copy_advice(|| "a_hi", region, a_4, abcd_row)?;
37        let a_value = a.dense_halves.value();
38        let a_cell = region.assign_advice(
39            || "a",
40            a_5,
41            abcd_row,
42            || a_value.map(|a| pallas::Base::from(a as u64)),
43        )?;
44        let a = AssignedWord32 {
45            cell: a_cell,
46            value: a_value,
47        };
48
49        let b = self.assign_digest_word(region, abcd_row, a_6, a_7, a_8, b.dense_halves)?;
50        let c = self.assign_digest_word(region, abcd_row + 1, a_3, a_4, a_5, c.dense_halves)?;
51        let d = self.assign_digest_word(region, abcd_row + 1, a_6, a_7, a_8, d)?;
52
53        // Assign digest for E, F, G, H
54        e.dense_halves
55            .0
56            .copy_advice(|| "e_lo", region, a_3, efgh_row)?;
57        e.dense_halves
58            .1
59            .copy_advice(|| "e_hi", region, a_4, efgh_row)?;
60        let e_value = e.dense_halves.value();
61        let e_cell = region.assign_advice(
62            || "e",
63            a_5,
64            efgh_row,
65            || e_value.map(|e| pallas::Base::from(e as u64)),
66        )?;
67        let e = AssignedWord32 {
68            cell: e_cell,
69            value: e_value,
70        };
71
72        let f = self.assign_digest_word(region, efgh_row, a_6, a_7, a_8, f.dense_halves)?;
73        let g = self.assign_digest_word(region, efgh_row + 1, a_3, a_4, a_5, g.dense_halves)?;
74        let h = self.assign_digest_word(region, efgh_row + 1, a_6, a_7, a_8, h)?;
75
76        Ok([
77            a, b, c, d, e, f, g, h,
78        ])
79    }

81    fn assign_digest_word(
82        &self,
83        region: &mut Region<'_, pallas::Base>,
84        row: usize,
85        lo_col: Column<Advice>,
86        hi_col: Column<Advice>,
87        word_col: Column<Advice>,
88        dense_halves: RoundWordDense,
89    ) -> Result<AssignedWord32, Error> {
90        dense_halves.0.copy_advice(|| "lo", region, lo_col, row)?;
91        dense_halves.1.copy_advice(|| "hi", region, hi_col, row)?;
92
93        let value = dense_halves.value();
94        let cell = region.assign_advice(
95            || "word",
96            word_col,
97            row,
98            || value.map(|val| pallas::Base::from(val as u64)),
99        )?;
100
101        Ok(AssignedWord32 { cell, value })
102    }
